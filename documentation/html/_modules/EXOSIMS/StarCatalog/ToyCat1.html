# -*- coding: utf-8 -*-
import os, inspect
import numpy as np
import astropy.units as u
from astropy.coordinates import SkyCoord
from EXOSIMS.Prototypes.StarCatalog import StarCatalog


[docs]class ToyCat1(StarCatalog):
    """
    Toy Star Catalog class
    
    This class populates the star catalog used in EXOSIMS from
    a toy catalog generated by Rhonda Morgan, fed in as a CSV file based on 
    the original Excel spreadsheet.
    
    Attributes:
        Only StarCatalog prototype attributes are used.
    
    """
    
    def __init__(self,catalogpath=None,**specs):
        """
        Constructor for ToyCat1
        
        Args:
            catalogpath (string):
                Full path to catalog CSV file. Defaults to toy_star_list_102715v2.csv 
                in the StarCatalog folder.
        
        """
       
        if catalogpath is None:
            classpath = os.path.split(inspect.getfile(self.__class__))[0]
            filename = 'toy_star_list_102715v2.csv'
            catalogpath = os.path.join(classpath, filename)
        
        if not os.path.exists(catalogpath):
            raise IOError('Catalog File %s Not Found.'%catalogpath)
        
        #read csv file, contains 'Name','Spec','Vmag','L','parx(asec)','dist','ra','dec'
        data = np.genfromtxt(catalogpath, dtype=float, delimiter=",", skip_header = 1)
        
        StarCatalog.__init__(self, ntargs=len(data), **specs)
        
        # list of astropy attributes
        self.dist = data[:,-3]*u.pc
        self.parx = self.dist.to('mas',equivalencies=u.parallax())
        self.coords = SkyCoord(ra=data[:,-2]*u.deg, dec=data[:,-1]*u.deg, \
                distance=self.dist)
        
        # list of non-astropy attributes
        self.Name = data[:,0].astype(str)
        specTypes = np.array(['F','G','K','M'])
        self.Spec = specTypes[data[:,1].astype(int)-1]
        self.Vmag = data[:,2]
        self.L = data[:,3]
        self.MV = self.Vmag - 5*(np.log10(self.dist.value) - 1)
